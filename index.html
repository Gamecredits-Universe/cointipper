CoinTipper v0.1

Setup

Include cointipper.js along with jQuery, Bootstrap Modal, and stylesheet.

For a Bitcoin donate button:
bitcoinDonateButton('NAME', 'ADDRESS', 'LABEL');

For a Dogecoin donate button:
dogecoinDonateButton('NAME', 'ADDRESS', 'LABEL');

Example: 


<!-- Put buttons anywhere in your layout -->
<p>
    <span id="donate-bitcoin" class="donate-btn"></span>
    <span id="donate-dogecoin" class="donate-btn"></span>
</p>

<!-- Include scripts before closing <body> tag -->
<script src="//cdnjs.cloudflare.com/ajax/libs/jquery/2.0.2/jquery.min.js"></script>
<script src="//cdnjs.cloudflare.com/ajax/libs/bootstrap-modal/2.1.0/bootstrap-modal.pack.js"></script>
<script src="assets/js/cointipper.js"></script>

$(function() {
    dogecoinDonateButton('CoinTipper', 'DNuK9cSYUgM5YgyFfKThg8UbVLQS1DWvSS', 'CoinTipper Tip Jar');
    bitcoinDonateButton('CoinTipper', '1G967MwCf6XsMufpot3wHzbYnU3d2RevD5', 'CoinTipper Tip Jar');
});
